def no_tilde(x):
    tilds = ['á', 'é', 'í' ,'ó' ,'ú', '"', ',', '.']
    new = ['a', 'e', 'i','o','u', '', '', '']
    for i, t in enumerate(tilds):
        if t in x:
            x = x.replace(t, new[i])
    return x

def stem(x):
    wlist =  [SnowballStemmer("spanish").stem(w) for w in x.split()]
    return ' '.join(word for word in wlist)

def clean_desc(x, stopwords):
    words = x.split()
    new_words = [word for word in words if word not in stopwords]
    return ' '.join(word.lower() for word in new_words)

def plot_by_cont_filter(df, factor, word, title):
    '''
    Inputs:
    - df (pd.DataFrame)
    - factor (col)
    - words (list)
    - title (str)
    '''
    
    df_gmth = pd.DataFrame({'Count' : df.groupby(['mnth_yr']).size()}).reset_index()
    df_int = pd.DataFrame({'Dum': df[factor].str.contains(word), 'mnth_yr': df['mnth_yr']})
    df_intc = df_int.groupby(['mnth_yr']).sum().reset_index()
    df_gmth['int'] = df_intc.Dum
    df_gmth['int_rat'] = df_gmth['int'] / df_gmth['Count']
    ax = df_gmth.int_rat.plot()
    plt.title(title)
    return df_gmth

III. Additional Columns¶

Add the following columns, based on description texts:

  • Weapon type (firegun, explosives, knife, machine-gun / large scale, unknown; can take multiple values)
  • Number of victims (integer)
  • Presence of torture (dummy)
  • Presence of threat (dummy)
  • Murder (dummy)
  • Infrastructure desctruction (dummy)

  • Type of victim:

    • Government official (dummy)
    • Civic leader (dummy)
    • Farmer (dummy)

  • Selective (presence of manually selected words, dummy)

def weapon(text, d):
    weaps= set()
    for weap, wlist in d.items():
        for w in wlist:
            if SnowballStemmer("spanish").stem(w) in text:
                weaps.add(weap)
    if len(weaps) == 0:
        return 'unknown'
    else:
        return str(list(weaps))

def dummify_cont(text, words):

def six_aside(S, words):
    d = {}
    txt = S.split()
    words = [SnowballStemmer("spanish").stem(wd) for wd in words]
    for wd in words:
        d[wd] = []
    for i, w in enumerate(txt):

        if w in words:
            
            d[w] += [txt[i+1]]
            d[w] += [txt[i+2]]
            d[w] += [txt[i-1]]
            d[w] += [txt[i-2]]
    return d

def sortFreqDict(freqdict):
    aux = [(freqdict[key], key) for key in freqdict]
    aux.sort()
    aux.reverse()
    return aux

def word_freq(d, stopwords):
    d_cnt = {}
    for k,v in d.items():
        d_cnt[k] = []
        d_counts={}
        for x in v:
            if x not in stopwords:
                d_counts[x] = v.count(x)
        d_cnt[k]= sortFreqDict(d_counts)
        
    return d_cnt
